176 public List<RangeInt> RTTCheck(string checkText)
177 {
178 //返す用のリスト
179 List<RangeInt> rangeDataList = new List<RangeInt>();
180
181 //全タグ分判定
182 foreach (var match in tagMatchDictionary)
183 {
184 //タグがあるかどうか判定
185 string tag = match.Value;
186 var result = Regex.Match(checkText, tag);
187
188 //判定が成功の限り無限ループ
189 while (result.Success)
190 {
191 //範囲設定
192 RangeInt range = new RangeInt(result.Index, result.Value.Length);
193 rangeDataList.Add(range);
194 //Debug.Log(tag + " is hit b:" + range.start + " e:" + range.end);
195
196 //次の判定
197 result = result.NextMatch();
198 }
199 }
200
201 return rangeDataList;
202 }
