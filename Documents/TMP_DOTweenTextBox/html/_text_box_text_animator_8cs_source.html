96 RangeInt beforeRange = new RangeInt(0,0);
97 foreach (RangeInt defaultAnimationRange in defaultAnimationRanges)

100 lengthSum += defaultAnimationRange.start - beforeRange.end;

110 lengthSum += defaultAnimationRange.length;

146 private List<RangeInt> GetDefaultRange(List<TMP_TagAnimationData> rangeList)
147 {
148 //返す用のリスト
149 List<RangeInt> result = new List<RangeInt>(100);
150
151 for (int beginIndex = 0; beginIndex < text.text.Length; beginIndex++)
152 {
153 bool isGenerate = false;
154 int end = beginIndex;
155 for (int endIndex = beginIndex + 1; endIndex < text.text.Length + 1; endIndex++)
156 {
157 isGenerate = true;
158
159 bool isRangeIn = false;
160 foreach (TMP_TagAnimationData rangeInt in rangeList)
161 {
162 //範囲内入ってたら範囲内フラグを立てる
163 if ((beginIndex > rangeInt.beginIndex && beginIndex < rangeInt.endIndex) ||
164 (endIndex > rangeInt.beginIndex && endIndex < rangeInt.endIndex) )
165 {
166 isRangeIn = true;
167 }
168 //終わりと始まりが同じなら生成フラグを立てない
169 if(endIndex == rangeInt.beginIndex)
170 {
171 isGenerate = false;
172 };
173 }
174
175 //生成する場合終わりは-1
176 if (isGenerate) end = endIndex - 1;
177 //範囲内フラグが立って入ればスキップ
178 if (isRangeIn) break;
179
180 //終わりがテキストの文字列の終わりなら生成
181 if (endIndex == text.text.Length)
182 {
183 isGenerate = true;
184 end = text.text.Length;
185 }
186 }
187
188 //生成フラグが立っていて開始位置が終了位置より前なら範囲データを生成
189 if (isGenerate && beginIndex < end)
190 {
191 //Generate
192 result.Add(new RangeInt(beginIndex,end - beginIndex));
193 beginIndex = end;
194 }
195 }
196
197 return result;
